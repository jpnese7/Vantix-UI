Quick Start

local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/user/repo/main/AntigravityLib.lua"))()

local Window = Library:CreateWindow({
    Name = "My Script Hub",
    LoadingTitle = "Antigravity UI",
    LoadingSubtitle = "by YourName"
})

local Tab = Window:CreateTab("Main")

Tab:CreateButton("Click Me", function()
    print("Hello World!")
end)

Tab:CreateToggle("Fly", false, function(Value)
    print("Fly is now: " .. tostring(Value))
end)
